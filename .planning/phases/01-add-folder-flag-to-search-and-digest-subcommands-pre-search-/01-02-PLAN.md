Add a new function before `runDigest` (around line 300, before the `function runDigest()` declaration):

```javascript
/**
 * Parse process.argv for digest subcommand arguments.
 * Expects: node outlook.js digest [--folder <name>]
 *
 * @param {string[]} argv - process.argv
 * @returns {{ folder: string|null }}
 */
function parseDigestArgs(argv) {
  const args = argv.slice(3); // skip node, outlook.js, digest
  let folder = null;

  for (let i = 0; i < args.length; i++) {
    if (args[i] === '--folder' && i + 1 < args.length) {
      folder = args[i + 1];
      i++; // skip next
    }
  }

  return { folder };
}
```
