```javascript
/**
 * Filter harmless warnings from agent-browser stderr.
 * The "--profile ignored: daemon already running" warning is expected on
 * subsequent calls when the daemon was started by a prior command.
 */
function filterStderr(stderr) {
  if (!stderr) return '';
  return stderr.split('\n')
    .filter(line => !line.includes('--profile ignored'))
    .join('\n')
    .trim();
}
```
