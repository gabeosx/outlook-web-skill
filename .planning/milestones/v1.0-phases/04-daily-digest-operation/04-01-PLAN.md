**Part A — Implement scoreMessage() in lib/digest.js:**

Add or complete the `scoreMessage(parsed, rawName)` function. It receives the parsed message object from parseAccessibleName() and the ORIGINAL (pre-suffix-stripped but post-prefix-stripped) accessible name string used to compute urgency keyword matches.

Scoring model (per D-14 through D-17):

```javascript
const URGENCY_KEYWORDS = [
  'urgent', 'asap', 'action required', 'immediate', 'deadline',
  'due today', 'overdue', 'eod', 'eow', 'critical', 'expires', 'time-sensitive'
];

function scoreMessage(parsed, searchText) {
  // searchText = concat of from + subject + preview (case-insensitive match target)
  const signals = [];
  let score = 0;

  // Unread: +40 points (D-14)
  if (!parsed.is_read) {
    score += 40;
    signals.push('unread');
  }

  // High-importance prefix in raw accessible name: +40 points (D-14)
  // The "Important " prefix is stripped by parseAccessibleName before returning,
  // but we need to detect it. Parse it from the searchText or pass rawName.
  // APPROACH: pass the accessible name BEFORE parseAccessibleName strips "Important " prefix.
  // In the accumulation loop: check rawName.includes('Important ') after stripping Unread prefix
  // but before stripping Important prefix.
  if (parsed._hadImportantPrefix) {  // set by wrapper logic in accumulation loop
    score += 40;
    signals.push('high_importance');
  }

  // Urgency keywords: +5 each, capped at +20 (D-14/D-15)
  const haystack = (searchText || '').toLowerCase();
  let keywordPoints = 0;
  for (const kw of URGENCY_KEYWORDS) {
    if (haystack.includes(kw)) {
      const points = Math.min(5, 20 - keywordPoints);
      if (points > 0) {
        keywordPoints += points;
        score += points;
        signals.push('keyword:' + kw);
      }
    }
    if (keywordPoints >= 20) break;
  }

  return { importance_score: score, importance_signals: signals };
}
```
