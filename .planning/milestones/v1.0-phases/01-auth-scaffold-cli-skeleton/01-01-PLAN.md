```javascript
'use strict';

/**
 * Write success JSON to stdout.
 * @param {string} operation
 * @param {any} results - array or null
 */
function outputOk(operation, results = null) {
  process.stdout.write(
    JSON.stringify({ operation, status: 'ok', results, error: null }) + '\n'
  );
}

/**
 * Write error JSON to stdout and exit(1).
 * @param {string} operation
 * @param {string} code - one of: INVALID_ARGS, SESSION_INVALID, AUTH_REQUIRED, OPERATION_FAILED
 * @param {string} message
 */
function outputError(operation, code, message) {
  process.stdout.write(
    JSON.stringify({ operation, status: 'error', results: null, error: { code, message } }) + '\n'
  );
  process.exit(1);
}

/**
 * Write diagnostic message to stderr only. Never touches stdout.
 * @param {string} msg
 */
function log(msg) {
  process.stderr.write(`[outlook-skill] ${msg}\n`);
}

module.exports = { outputOk, outputError, log };
```

**outlook.js** — Entry point. Exact implementation:

```javascript
'use strict';
const os = require('os');
const { outputError } = require('./lib/output');

// ── Step 1: Env var validation (before argv, before any browser) ──
const REQUIRED_ENV = ['OUTLOOK_BASE_URL', 'OUTLOOK_BROWSER_PATH', 'OUTLOOK_BROWSER_PROFILE'];
for (const key of REQUIRED_ENV) {
  if (!process.env[key]) {
    outputError('startup', 'INVALID_ARGS', `Missing required env var: ${key}`);
    // outputError exits — code below this loop never runs for missing vars
  }
}

// ── Step 2: Normalize tilde in path env vars ──
function expandTilde(p) {
  return typeof p === 'string' ? p.replace(/^~(?=$|\/|\\)/, os.homedir()) : p;
}
process.env.OUTLOOK_BROWSER_PATH = expandTilde(process.env.OUTLOOK_BROWSER_PATH);
process.env.OUTLOOK_BROWSER_PROFILE = expandTilde(process.env.OUTLOOK_BROWSER_PROFILE);

// ── Step 3: Extract allowed domain for SAFE-02 ──
let allowedDomain;
try {
  allowedDomain = new URL(process.env.OUTLOOK_BASE_URL).hostname;
} catch {
  outputError('startup', 'INVALID_ARGS', `OUTLOOK_BASE_URL is not a valid URL: ${process.env.OUTLOOK_BASE_URL}`);
}

// ── Step 4: Subcommand dispatch ──
const VALID_COMMANDS = ['auth', 'search', 'read', 'digest'];
const cmd = process.argv[2];

if (!cmd || !VALID_COMMANDS.includes(cmd)) {
  outputError(cmd ?? 'unknown', 'INVALID_ARGS', `Unknown subcommand: "${cmd ?? ''}". Valid: ${VALID_COMMANDS.join(', ')}`);
}

// Export allowedDomain for use by lib/run.js and subcommand handlers
module.exports = { allowedDomain };

switch (cmd) {
  case 'auth':
    require('./lib/session').runAuth();
    break;
  case 'search':
    outputError('search', 'OPERATION_FAILED', 'search not yet implemented — coming in Phase 2');
    break;
  case 'read':
    outputError('read', 'OPERATION_FAILED', 'read not yet implemented — coming in Phase 3');
    break;
  case 'digest':
    outputError('digest', 'OPERATION_FAILED', 'digest not yet implemented — coming in Phase 4');
    break;
}
```

Create `lib/run.js` — the single point where agent-browser is invoked. All safety env vars are set here per invocation. Never mutate `process.env` globally.

Exact implementation:

```javascript
'use strict';
const path = require('path');
const { spawnSync } = require('child_process');
const { log } = require('./output');

// Resolve policy file paths relative to the project root (where outlook.js lives)
const PROJECT_ROOT = path.resolve(__dirname, '..');

/**
 * Invoke agent-browser as a subprocess.
 * Sets all AGENT_BROWSER_* safety env vars on the child environment.
 * Never mutates process.env.
 *
 * @param {string[]} args - Arguments passed to agent-browser after the standard flags
 * @param {string} policyFile - Relative path to policy JSON (e.g., 'policy-auth.json')
 * @param {Object} [opts] - Optional overrides
 * @param {boolean} [opts.headed] - If true, pass --headed flag (auth only)
 * @returns {{ stdout: string, stderr: string, status: number }}
 */
function runAgentBrowser(args, policyFile, opts = {}) {
  const allowedDomain = process.env.OUTLOOK_BASE_URL
    ? new URL(process.env.OUTLOOK_BASE_URL).hostname
    : '';

  const policyPath = path.join(PROJECT_ROOT, policyFile);

  const childEnv = {
    ...process.env,
    AGENT_BROWSER_ACTION_POLICY: policyPath,
    AGENT_BROWSER_ALLOWED_DOMAINS: allowedDomain,
    AGENT_BROWSER_CONTENT_BOUNDARIES: '1',
    AGENT_BROWSER_EXECUTABLE_PATH: process.env.OUTLOOK_BROWSER_PATH,
    AGENT_BROWSER_PROFILE: process.env.OUTLOOK_BROWSER_PROFILE,
  };

  // --headed is only passed for interactive auth — never set globally
  const headedFlag = opts.headed ? ['--headed'] : [];

  const finalArgs = [
    '-q',
    '--session-name', 'outlook-skill',
    ...headedFlag,
    ...args,
  ];

  log(`agent-browser ${finalArgs.join(' ')}`);

  const result = spawnSync('agent-browser', finalArgs, {
    cwd: PROJECT_ROOT,
    env: childEnv,
    stdio: ['ignore', 'pipe', 'pipe'],
    encoding: 'utf8',
  });

  if (result.stderr) {
    log(`stderr: ${result.stderr.trim()}`);
  }

  return {
    stdout: result.stdout || '',
    stderr: result.stderr || '',
    status: result.status ?? 1,
  };
}

/**
 * Parse agent-browser --json output safely.
 * Returns null if stdout is not valid JSON or success is false.
 * @param {string} stdout
 * @returns {any|null}
 */
function parseAgentBrowserJson(stdout) {
  try {
    const parsed = JSON.parse(stdout.trim());
    if (!parsed.success) return null;
    return parsed;
  } catch {
    return null;
  }
}

module.exports = { runAgentBrowser, parseAgentBrowserJson };
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| CLI args → outlook.js | process.argv[2] is untrusted user input that must be validated before dispatch |
| env vars → outlook.js | OUTLOOK_BASE_URL, OUTLOOK_BROWSER_PATH, OUTLOOK_BROWSER_PROFILE are trusted (user-configured) but must be validated for format/presence |
| outlook.js → agent-browser subprocess | Policy file path passed via env var controls what the browser can do; must be set before spawn |
| agent-browser stdout → outlook.js | agent-browser output is treated as structured data; must be parsed defensively |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-01 | Tampering | policy-auth.json / policy-read.json | mitigate | Files live in the project directory owned by the user; no runtime writes to these files. Policy is read by agent-browser at spawn time via absolute path resolved in lib/run.js (`path.join(PROJECT_ROOT, policyFile)`), so relative path traversal in policyFile argument is blocked. |
| T-01-02 | Elevation of Privilege | lib/run.js — policyFile parameter | mitigate | `runAgentBrowser` accepts a policyFile string internally from outlook.js dispatch logic (not from user input). `search`, `read`, `digest` always receive `'policy-read.json'`; only `auth` receives `'policy-auth.json'`. User cannot pass a different policy path via CLI args. |
| T-01-03 | Information Disclosure | agent-browser stdout forwarded to calling agent | mitigate | AGENT_BROWSER_CONTENT_BOUNDARIES=1 is set on all subprocess invocations (lib/run.js). agent-browser stdout is captured via stdio pipe, never forwarded to process.stdout directly — only the constructed JSON envelope is written to stdout. |
| T-01-04 | Denial of Service | OUTLOOK_BASE_URL malformed URL → URL constructor throws | mitigate | outlook.js wraps `new URL(process.env.OUTLOOK_BASE_URL)` in try/catch and returns INVALID_ARGS JSON error rather than crashing with an uncaught exception. |
| T-01-05 | Spoofing | Missing/empty env vars causing undefined behavior | mitigate | REQUIRED_ENV validation loop runs before any code that uses env var values. Missing var triggers outputError('startup', 'INVALID_ARGS', ...) which calls process.exit(1). |
| T-01-06 | Tampering | process.env global mutation contaminating subprocess env | mitigate | lib/run.js creates a merged `childEnv` object per invocation using `{...process.env, ...overrides}`. No `process.env.AGENT_BROWSER_*` assignments anywhere in the codebase. |
</threat_model>
